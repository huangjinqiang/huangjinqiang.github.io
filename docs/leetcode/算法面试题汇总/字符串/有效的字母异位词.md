[有效的字母异位词](https://leetcode-cn.com/leetbook/read/top-interview-questions/xar9lv/)


### 题意
* 只包含小写字母

### 代码
```js
/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var isAnagram = function(s, t) {
  const { length } = s
  if (length !== t.length) return false

  const memo = {}
  for (let i = 0; i < length; i += 1) {
    if (memo[s[i]]) {
      memo[s[i]] += 1
    } else {
      memo[s[i]] = 1
    }
  }

  for (let i = 0; i < length; i += 1) {
    if (!memo[t[i]]) {
      return false
    }
    memo[t[i]] -= 1
  }

  return true
};
```
