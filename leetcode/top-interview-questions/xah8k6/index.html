验证回文串 (opens new window)

# 题意

  • 只考虑字母和数字字符，可以忽略字母的大小写
  • 将空字符串定义为有效的回文串

# 代码

/**
 * @param {string} s
 * @return {boolean}
 */
var isPalindrome = function(s) {
  const { length } = s
  if (length === 0) { return true } // 特判

  let start = 0;
  let end = length - 1
  const reg = /[a-zA-Z0-9]/
  while (true) {
    while (start < end) {
      if (reg.test(s[start])) {
        break
      }
      start += 1
    }
    while (start < end) {
      if (reg.test(s[end])) {
        break
      }
      end -= 1
    }

    if (start >= end) {
      return true
    }

    if (s[start].toLowerCase() !== s[end].toLowerCase()) {
      return false
    }

    start += 1
    end -= 1
  }

  return true
}
